// HÀM RENDER KHÓA HỌC VÀO TRANG CHỦ (Bổ sung hàm bị thiếu) function renderCourseCards() { const grid = document.getElementById('courseGrid'); grid.innerHTML = ''; // Làm sạch grid trước khi render courseConfig.forEach(course => { // Xử lý hiển thị link: Có link thì hiện nút, không có thì báo "Đang cập nhật" const linkHTML = course.link ? ` Vào giảng dạy ` : `Đang cập nhật...`; const sessionHTML = course.sessions > 0 ? `${course.sessions} Sessions` : ``; // Thẻ rỗng để đẩy linkHTML sang phải nhờ justify-between // Template HTML cho từng Card (áp dụng .glass-panel và .hover-5d) grid.innerHTML += `
${course.label}

${course.title}

${course.desc}

${sessionHTML} ${linkHTML}
`; }); }
